//Сокращение дроби
DM_Q RED_Q_Q(DM_Q f)
{
	DM_Q res;  //Конечный вид дроби
	DM_N os;

	os = GCF_NN_N(ABS_Z_N(f.a), ABS_Z_N(f.b)); //Поиск НОД числителя и знаменателя
	res.a = DIV_ZZ_Z(f.a, os); //Сокращение на НОД числитель
	res.b = DIV_ZZ_Z(f.b, os); //Сокращение на НОД знаменатель

	return res;
}
