DM_P DM_P::operator/(DM_P pDivider) //Частное от деления многочлена на многочлен при делении с остатком
{
	DM_P pQuotient,
		 pBuf,
		 pEnd;
	pBuf.c = new DM_Q[m];
	pQuotient.c = new DM_Q[m];
	pEnd.c = c;
	pEnd.m = m;
	while (pEnd.DEG_PP_N() >= pDivider.DEG_PP_N())
	{
		pBuf.m = 0;
		pBuf.c[0] = pEnd.c[0].DIV_QQ_Q(pDivider.c[0]);
		pBuf = pBuf.MUL_Pxk_P(pEnd.DEG_PP_N() - pDivider.DEG_PP_N());	//домножаем полученное частное коэффициентов на X в степени разности степеней остатка и делителя
		pQuotient = pQuotient.ADD_PP_P(pBuf);
		pEnd = pEnd.SUB_PP_P(pBuf.MUL_PP_P(pDivider));	//вычитаем из промежуточного результата произведение буфера и делителя
	}
	delete pBuf.c;
	return pQuotient;	//возврат полученного частного	
}
