//Умножение целых чисел
DM_Z MUL_ZZ_Z(DM_Z n1, DM_Z n2)
{
	DM_N n1abs;			//Модуль 1го числа
	DM_N n2abs;			//Модуль 2го
	DM_N natres;			//Модуль произведения
	DM_Z res;			//Произвеление
	n1abs = ABS_Z_N(n1);
	n2abs = ABS_Z_N(n2);
	natres = MUL_NN_N(n1, n2);
	res = TRANS_N_Z(natres);	
	if(n1.b == n2.b){		//Если знаки чисел равны – производим произведение
		return res;
	}else{			//Если знаки чисел не равны – домножим произведение на -1
		return MUL_ZM_Z(res)	
	}	
}
